\[ N(A \cup B) + 8 = 30 \]

\[ 22 = 15 + 10 - N(A \cap B) \]
\[ N(A \cap B) = 3 \]

\[ N(A \cap B^c) = N(A) - N(A \cap B) \]
\[ N(A \cap B^c) = 15 - 3 = 12 \]
